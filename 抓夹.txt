### 第 1 份代码：生成方块与物理世界的 Launch 文件
我们将自动生成的小方块坐标改到 `X=0.45, Y=0.0`。

打开 `~/dummy_ws/src/dummy-ros2_description/launch/gazebo.launch.py`，全选替换为：

```python
import os
import xacro
from ament_index_python.packages import get_package_share_directory
from launch import LaunchDescription
from launch.actions import IncludeLaunchDescription, TimerAction
from launch.launch_description_sources import PythonLaunchDescriptionSource
from launch.substitutions import PathJoinSubstitution
from launch_ros.actions import Node
from launch_ros.substitutions import FindPackageShare

def generate_launch_description():
    share_dir = get_package_share_directory('dummy-ros2_description')
    xacro_file = os.path.join(share_dir, 'urdf', 'dummy-ros2.xacro')
    robot_description_config = xacro.process_file(xacro_file)
    robot_urdf = robot_description_config.toxml()

    robot_state_publisher_node = Node(
        package='robot_state_publisher',
        executable='robot_state_publisher',
        parameters=[{'robot_description': robot_urdf, 'use_sim_time': True}]
    )

    gazebo = IncludeLaunchDescription(
        PythonLaunchDescriptionSource([PathJoinSubstitution([FindPackageShare('gazebo_ros'), 'launch', 'gazebo.launch.py'])]),
    )

    spawn_entity = Node(
        package='gazebo_ros',
        executable='spawn_entity.py',
        arguments=['-entity', 'dummy-ros2', '-topic', 'robot_description']
    )
    delayed_spawn = TimerAction(period=5.0, actions=[spawn_entity])

    # 🌟 核心：将蓝块生成在机械臂的正前方 0.45 米处（黄金抓取区）
    box_urdf_path = os.path.join(share_dir, 'urdf', 'target_box.urdf')
    spawn_box = Node(
        package='gazebo_ros',
        executable='spawn_entity.py',
        arguments=['-entity', 'target_box', '-file', box_urdf_path, '-x', '0.45', '-y', '0.0', '-z', '0.02']
    )
    delayed_spawn_box = TimerAction(period=6.0, actions=[spawn_box])

    load_joint_state_broadcaster = Node(package="controller_manager", executable="spawner", arguments=["joint_state_broadcaster"])
    load_arm_controller = Node(package="controller_manager", executable="spawner", arguments=["arm_controller"])
    load_gripper_controller = Node(package="controller_manager", executable="spawner", arguments=["gripper_controller"])

    return LaunchDescription([
        robot_state_publisher_node, gazebo,
        delayed_spawn, delayed_spawn_box,
        TimerAction(period=7.0, actions=[load_joint_state_broadcaster]),
        TimerAction(period=8.0, actions=[load_arm_controller]),
        TimerAction(period=8.5, actions=[load_gripper_controller]),
    ])
```

### 第 2 份代码：AI 视觉识别的 Python 文件
这段代码增强了颜色抗干扰能力，并且精准还原了相机到物理世界的坐标系映射。

打开 `~/dummy_ws/src/dummy_grasping/src/vision_grasp.py`，全选替换为：

```python
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from geometry_msgs.msg import TransformStamped
from tf2_ros import TransformBroadcaster
from cv_bridge import CvBridge
import cv2
import numpy as np

class VisionGraspNode(Node):
    def __init__(self):
        super().__init__('vision_grasp_node')
        self.bridge = CvBridge()
        self.latest_depth_image = None
        self.tf_broadcaster = TransformBroadcaster(self)

        self.depth_sub = self.create_subscription(Image, '/camera/rgbd_camera/depth/image_raw', self.depth_callback, 10)
        self.color_sub = self.create_subscription(Image, '/camera/rgbd_camera/image_raw', self.color_callback, 10)
        self.get_logger().info('🟢 视觉节点启动，正在寻找黄金区域的小蓝块...')

    def depth_callback(self, msg):
        try:
            self.latest_depth_image = self.bridge.imgmsg_to_cv2(msg, "32FC1")
        except:
            pass

    def color_callback(self, msg):
        if self.latest_depth_image is None: return
        try:
            cv_image = self.bridge.imgmsg_to_cv2(msg, "bgr8")
            h, w, _ = cv_image.shape
            
            hsv = cv2.cvtColor(cv_image, cv2.COLOR_BGR2HSV)
            # 严格锁定蓝色，防止识别阴影
            mask = cv2.inRange(hsv, np.array([100, 100, 50]), np.array([140, 255, 255]))
            contours, _ = cv2.findContours(mask, cv2.RETR_TREE, cv2.CHAIN_APPROX_SIMPLE)
            
            u, v = int(w/2), int(h/2)
            target_found = False

            if len(contours) > 0:
                largest_contour = max(contours, key=cv2.contourArea)
                if cv2.contourArea(largest_contour) > 80:
                    M = cv2.moments(largest_contour)
                    if M["m00"] > 0:
                        u, v = int(M["m10"] / M["m00"]), int(M["m01"] / M["m00"])
                        target_found = True
                        x_box, y_box, w_box, h_box = cv2.boundingRect(largest_contour)
                        cv2.rectangle(cv_image, (x_box, y_box), (x_box + w_box, y_box + h_box), (0, 255, 255), 2)

            z_depth = self.latest_depth_image[v, u]
            fx, fy, cx, cy = 320.0, 320.0, 320.0, 240.0

            if not np.isnan(z_depth) and z_depth > 0.0:
                x_cv = (u - cx) * z_depth / fx
                y_cv = (v - cy) * z_depth / fy
                
                # 转换到 ROS 坐标系
                ros_x, ros_y, ros_z = float(z_depth), float(-x_cv), float(-y_cv)
                
                if target_found:
                    t = TransformStamped()
                    t.header.stamp = self.get_clock().now().to_msg()
                    t.header.frame_id = 'camera_link' # 必须与 URDF 中的相机名字一致
                    t.child_frame_id = 'target_object'
                    t.transform.translation.x = ros_x
                    t.transform.translation.y = ros_y
                    t.transform.translation.z = ros_z
                    t.transform.rotation.w = 1.0
                    self.tf_broadcaster.sendTransform(t)

                color = (0, 255, 0) if target_found else (0, 165, 255)
                text = f"Locked: (X:{ros_x:.2f}, Y:{ros_y:.2f}, Z:{ros_z:.2f})" if target_found else "Scanning..."
                cv2.putText(cv_image, text, (20, 40), cv2.FONT_HERSHEY_SIMPLEX, 0.7, color, 2)
            
            cv2.drawMarker(cv_image, (u, v), (0, 0, 255), cv2.MARKER_CROSS, 20, 2)
            cv2.imshow("Dummy Robot - Vision AI", cv_image)
            cv2.waitKey(1)
        except Exception as e:
            pass

def main(args=None):
    rclpy.init(args=args)
    node = VisionGraspNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

### 第 3 份代码：运动执行与抓取的 C++ 文件
不再强加别扭的姿态约束，完全基于空间位置规划，并留出了足够的抓夹长度补偿。

打开 `~/dummy_ws/src/dummy_grasping/src/grasp_executor.cpp`，全选替换为：

```cpp
#include <rclcpp/rclcpp.hpp>
#include <moveit/move_group_interface/move_group_interface.h>
#include <tf2_ros/transform_listener.h>
#include <tf2_ros/buffer.h>
#include <geometry_msgs/msg/transform_stamped.hpp>

int main(int argc, char * argv[])
{
  rclcpp::init(argc, argv);
  auto node = std::make_shared<rclcpp::Node>("grasp_executor", rclcpp::NodeOptions().automatically_declare_parameters_from_overrides(true));
  rclcpp::Logger logger = node->get_logger();
  RCLCPP_INFO(logger, "====== 🚀 终极流水线执行节点已启动 ======");

  moveit::planning_interface::MoveGroupInterface arm_group(node, "dummy_arm");
  moveit::planning_interface::MoveGroupInterface gripper_group(node, "gripper");

  tf2_ros::Buffer tf_buffer(node->get_clock());
  tf2_ros::TransformListener tf_listener(tf_buffer);
  geometry_msgs::msg::TransformStamped t_stamped;

  // 等待 AI 视觉发送坐标
  while (rclcpp::ok()) {
    try {
      t_stamped = tf_buffer.lookupTransform("base_link", "target_object", tf2::TimePointZero);
      break;
    } catch (const tf2::TransformException & ex) {
      rclcpp::sleep_for(std::chrono::milliseconds(500));
    }
  }

  double target_x = t_stamped.transform.translation.x;
  double target_y = t_stamped.transform.translation.y;
  double target_z = t_stamped.transform.translation.z;
  RCLCPP_INFO(logger, "🎯 计算得出真实抓取坐标: X: %.3f, Y: %.3f, Z: %.3f", target_x, target_y, target_z);

  // 确保目标在合理的工作范围内
  if (target_x < 0.2 || target_x > 0.6) {
    RCLCPP_ERROR(logger, "❌ 目标距离超出最佳抓取范围，请将物体移近或移远！");
    rclcpp::shutdown();
    return 0;
  }

  RCLCPP_INFO(logger, ">> 动作 1: 张开抓夹...");
  gripper_group.setNamedTarget("open");
  gripper_group.move();

  RCLCPP_INFO(logger, ">> 动作 2: 移动到方块上方 (安全悬停)...");
  // 使用 setPositionTarget 让 MoveIt 自己决定最顺畅的手臂姿态，高度离地 20 厘米
  arm_group.setPositionTarget(target_x, target_y, target_z + 0.20); 
  arm_group.move();

  RCLCPP_INFO(logger, ">> 动作 3: 笔直下降...");
  // 下降到指尖能夹住方块的位置 (假设手腕到指尖距离约 12 厘米，所以停在 0.12)
  arm_group.setPositionTarget(target_x, target_y, target_z + 0.12);
  arm_group.move();

  RCLCPP_INFO(logger, ">> 动作 4: 闭合抓夹...");
  gripper_group.setNamedTarget("close");
  gripper_group.move();

  RCLCPP_INFO(logger, ">> 动作 5: 成功抓取，抬起回原位...");
  arm_group.setNamedTarget("home");
  arm_group.move();

  RCLCPP_INFO(logger, "====== 🎉 完美通关！ ======");
  rclcpp::shutdown();
  return 0;
}
```
