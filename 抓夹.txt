下面，我们用 C++ 编写这最后一段执行代码。只需简单的三步：

### 第一步：编写 C++ 抓取执行节点

1. 在终端中，进入你模型包的 `src` 文件夹（如果没有就新建一个）：
   ```bash
   mkdir -p ~/dummy_ws/src/dummy-ros2_description/src
   cd ~/dummy_ws/src/dummy-ros2_description/src
   ```
2. 新建并打开文件 `grasp_executor.cpp`：
   ```bash
   gedit grasp_executor.cpp
   ```
3. **将以下“执行层”完整代码粘贴进去并保存：**
   *(这段代码会自动监听 Python 发出的 `target_object` 坐标，计算它相对于机械臂底座的绝对物理距离，并调用 MoveIt2 生成平滑抓取轨迹。)*

```cpp
#include <rclcpp/rclcpp.hpp>
#include <moveit/move_group_interface/move_group_interface.h>
#include <tf2_ros/transform_listener.h>
#include <tf2_ros/buffer.h>
#include <geometry_msgs/msg/pose_stamped.hpp>

int main(int argc, char * argv[])
{
  rclcpp::init(argc, argv);
  // 配置节点以自动从 MoveIt 提取参数
  auto node = std::make_shared<rclcpp::Node>("grasp_executor", rclcpp::NodeOptions().automatically_declare_parameters_from_overrides(true));
  rclcpp::Logger logger = node->get_logger();
  
  RCLCPP_INFO(logger, "====== 🚀 视觉抓取执行节点已启动 ======");

  // 1. 初始化 MoveIt 控制接口 (对应我们 SRDF 里配置的组名)
  moveit::planning_interface::MoveGroupInterface arm_group(node, "dummy_arm");
  moveit::planning_interface::MoveGroupInterface gripper_group(node, "gripper");

  // 2. 初始化 TF 空间监听器
  tf2_ros::Buffer tf_buffer(node->get_clock());
  tf2_ros::TransformListener tf_listener(tf_buffer);

  RCLCPP_INFO(logger, "📡 正在监听 AI 视觉节点发送的 [target_object] 坐标...");
  geometry_msgs::msg::TransformStamped transformStamped;
  
  // 持续监听，直到成功获取到 Python 算出来的方块坐标
  while (rclcpp::ok()) {
    try {
      // 核心计算：把相机的视野坐标，转换成机械臂底座(base_link)认识的物理坐标
      transformStamped = tf_buffer.lookupTransform("base_link", "target_object", tf2::TimePointZero);
      break;
    } catch (const tf2::TransformException & ex) {
      rclcpp::sleep_for(std::chrono::milliseconds(500));
    }
  }

  // 提取目标在真实物理世界中的 X, Y, Z
  double target_x = transformStamped.transform.translation.x;
  double target_y = transformStamped.transform.translation.y;
  double target_z = transformStamped.transform.translation.z;
  RCLCPP_INFO(logger, "🎯 锁定目标真实坐标: X: %.3f, Y: %.3f, Z: %.3f", target_x, target_y, target_z);

  // 3. 开始执行流水线抓取动作！
  RCLCPP_INFO(logger, ">> 动作 1: 张开抓夹...");
  gripper_group.setNamedTarget("open");
  gripper_group.move();

  RCLCPP_INFO(logger, ">> 动作 2: 移动到方块上方 (预抓取点)...");
  // 采用逆运动学最优解：只限制位置到达方块上方 15 厘米处，姿态由 AI 自由规划
  arm_group.setPositionTarget(target_x, target_y, target_z + 0.15); 
  arm_group.move();

  RCLCPP_INFO(logger, ">> 动作 3: 直线下降...");
  // 下降至刚好贴近物体的高度
  arm_group.setPositionTarget(target_x, target_y, target_z + 0.03);
  arm_group.move();

  RCLCPP_INFO(logger, ">> 动作 4: 闭合抓夹...");
  gripper_group.setNamedTarget("close");
  gripper_group.move();

  RCLCPP_INFO(logger, ">> 动作 5: 抓取完成，抬起回原位...");
  arm_group.setNamedTarget("home");
  arm_group.move();

  RCLCPP_INFO(logger, "====== 🎉 实验圆满结束！ ======");

  rclcpp::shutdown();
  return 0;
}
```

### 第二步：配置编译文件 (告诉系统如何编译 C++)

因为加入了 C++ 代码，我们需要在 `CMakeLists.txt` 中引入 ROS2 的规划库。
请打开 `~/dummy_ws/src/dummy-ros2_description/CMakeLists.txt`，将内容**完全替换**为：

```cmake
cmake_minimum_required(VERSION 3.8)
project(dummy-ros2_description)

if(CMAKE_COMPILER_IS_GNUCXX OR CMAKE_CXX_COMPILER_ID MATCHES "Clang")
  add_compile_options(-Wall -Wextra -Wpedantic)
endif()

# 寻找所有需要的 ROS2 依赖库
find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(moveit_ros_planning_interface REQUIRED)
find_package(tf2_ros REQUIRED)
find_package(geometry_msgs REQUIRED)

# 告诉编译器将 URDF 和模型复制过去
install(
  DIRECTORY meshes urdf launch
  DESTINATION share/${PROJECT_NAME}
)

# ======== 核心：编译刚才写的 C++ 抓取执行文件 ========
add_executable(grasp_executor src/grasp_executor.cpp)
ament_target_dependencies(grasp_executor rclcpp moveit_ros_planning_interface tf2_ros geometry_msgs)
install(TARGETS grasp_executor DESTINATION lib/${PROJECT_NAME})
# ====================================================

ament_package()
```
